#include <stdio.h>
#include <fcntl.h>
#include <unistd.h>

int main() {
    int fd;
    char buffer[50];

    fd = open("file.txt", O_CREAT | O_RDWR, 0777);
    write(fd, "Hello OS Lab", 12);

    lseek(fd, 0, SEEK_SET);
    read(fd, buffer, 12);
    buffer[12] = '\0';

    printf("File Content: %s\n", buffer);
    close(fd);

    return 0;
}

OUTPUT:
File Content: Hello OS Lab
