#include <stdio.h>
#include <fcntl.h>
#include <unistd.h>

int main() {
    int src = open("a.txt", O_RDONLY);
    int dest = open("b.txt", O_WRONLY | O_CREAT, 0644);
    char buf[100];
    int n;
    while((n = read(src, buf, 100)) > 0)
        write(dest, buf, n);
    close(src);
    close(dest);
    return 0;
}
