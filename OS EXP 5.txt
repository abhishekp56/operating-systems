#include <stdio.h>

int main() {
    int bt[10], pr[10], n, i, j, t;
    scanf("%d",&n);
    for(i=0;i<n;i++) scanf("%d%d",&bt[i],&pr[i]);
    for(i=0;i<n;i++)
        for(j=i+1;j<n;j++)
            if(pr[i]>pr[j]) {
                t=pr[i]; pr[i]=pr[j]; pr[j]=t;
                t=bt[i]; bt[i]=bt[j]; bt[j]=t;
            }
    for(i=0;i<n;i++) printf("%d %d\n",bt[i],pr[i]);
    return 0;
}
