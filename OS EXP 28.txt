#include <stdio.h>
#include <string.h>

int main() {
    FILE *fp;
    char line[100];
    char word[] = "Hello";

    fp = fopen("test.txt", "r");

    while(fgets(line, sizeof(line), fp)) {
        if(strstr(line, word)) {
            printf("%s", line);
        }
    }

    fclose(fp);
    return 0;
}
