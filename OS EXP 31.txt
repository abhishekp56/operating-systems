#include <stdio.h>

int main() {
    int frames[3], pages[] = {1,2,3,4,1,2,5};
    int i,j,k=0,fault=0,flag;

    for(i=0;i<3;i++)
        frames[i]=-1;

    for(i=0;i<7;i++) {
        flag=0;
        for(j=0;j<3;j++)
            if(frames[j]==pages[i])
                flag=1;

        if(flag==0) {
            frames[k]=pages[i];
            k=(k+1)%3;
            fault++;
        }
    }

    printf("Total Page Faults = %d\n", fault);
}

OUTPUT:
Total Page Faults = 7
