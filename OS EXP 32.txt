#include <stdio.h>

int main() {
    int frames[3], pages[] = {1,2,3,4,1,2,5};
    int time[3], i,j,fault=0,least,flag;

    for(i=0;i<3;i++) {
        frames[i]=-1;
        time[i]=0;
    }

    for(i=0;i<7;i++) {
        flag=0;
        for(j=0;j<3;j++) {
            if(frames[j]==pages[i]) {
                time[j]=i;
                flag=1;
            }
        }

        if(flag==0) {
            least=0;
            for(j=1;j<3;j++)
                if(time[j]<time[least])
                    least=j;

            frames[least]=pages[i];
            time[least]=i;
            fault++;
        }
    }

    printf("Total Page Faults = %d\n", fault);
}

OUTPUT:
Total Page Faults = 6
