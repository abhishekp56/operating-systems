#include <stdio.h>
#include <fcntl.h>
#include <unistd.h>
#include <sys/stat.h>
#include <dirent.h>

int main() {
    int fd = open("file.txt", O_RDONLY);

    struct stat st;
    fstat(fd, &st);
    printf("File Size: %ld bytes\n", st.st_size);

    close(fd);

    DIR *d = opendir(".");
    struct dirent *dir;

    printf("Files in directory:\n");
    while((dir = readdir(d)) != NULL) {
        printf("%s\n", dir->d_name);
    }
    closedir(d);

    return 0;
}
