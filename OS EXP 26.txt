#include <stdio.h>

int main() {
    FILE *fp;
    char text[50];

    fp = fopen("test.txt", "w");
    fprintf(fp, "Hello World\n");
    fclose(fp);

    fp = fopen("test.txt", "r");
    fgets(text, sizeof(text), fp);
    printf("File Content: %s", text);
    fclose(fp);

    return 0;
}

OUTPUT:
File Content: Hello World
