#include <stdio.h>

int main() {
    int blocks, i;
    printf("Enter number of blocks: ");
    scanf("%d", &blocks);

    int index[blocks];

    printf("Enter block numbers:\n");
    for(i = 0; i < blocks; i++)
        scanf("%d", &index[i]);

    printf("Index Block:\n");
    for(i = 0; i < blocks; i++)
        printf("Block %d -> %d\n", i, index[i]);

    return 0;
}

OUTPUT:
Enter number of blocks: 3
Enter block numbers:
10 25 40
Index Block:
Block 0 -> 10
Block 1 -> 25
Block 2 -> 40
