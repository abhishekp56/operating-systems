#include <stdio.h>
#include <sys/ipc.h>
#include <sys/msg.h>
#include <string.h>

struct msg { long type; char text[50]; };

int main() {
    int id = msgget(1234, 0666|IPC_CREAT);
    struct msg m;
    m.type=1;
    strcpy(m.text,"Hi");
    msgsnd(id,&m,sizeof(m.text),0);
    msgrcv(id,&m,sizeof(m.text),1,0);
    printf("%s",m.text);
    return 0;
}
